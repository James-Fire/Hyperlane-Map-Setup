	allow = { hidden_trigger = { always = yes } }
	effect = { }

generate_hyperlane_layout = {
	potential = { hidden_trigger = { always = yes } }
	allow = { hidden_trigger = { always = yes } }
	effect = { #Use the Current hyperlane options to generate the map
		if = {
			limit = {
				has_global_flag = generation_clusters
			} #Need to expand cluster generation to include the options present in the UI.
			#remove_global_flag = yes
		}
		else_if = {
			limit = {
				has_global_flag = generation_pie_slices
			}
			hms_generator_pie = yes
		}
		else_if = {
			limit = {
				has_global_flag = generation_corridor
			}
			hms_generator_closest_loop = yes
			#hms_generator_ugly_loop = yes
		}
		else_if = {
			limit = {
				has_global_flag = generation_center
			}
			hms_generator_star = yes
		}
		else_if = {
			limit = {
				has_global_flag = generation_circles
			}
			hms_generator_concentric_circles = yes
		}
		else_if = {
			limit = {
				has_global_flag = generation_all
			}
			hms_generator_full = yes
		}
	
	}
}

generation_formula = {
	potential = { hidden_trigger = { always = yes } }
	allow = { hidden_trigger = { always = yes } }
	effect = { #Swap between the various hyperlane layouts we've got.
		if = {
			limit = {
				has_global_flag = generation_clusters
			}
			remove_global_flag = generation_clusters
			set_global_flag = generation_pie_slices
		}
		else_if = {
			limit = {
				has_global_flag = generation_pie_slices
			}
			remove_global_flag = generation_pie_slices
			set_global_flag = generation_corridor
		}
		else_if = {
			limit = {
				has_global_flag = generation_corridor
			}

			set_global_flag = generation_center
		}
		else_if = {
			limit = {
				has_global_flag = generation_center
			}
			remove_global_flag = generation_center
			set_global_flag = generation_circles
		}
		else_if = {
			limit = {
				has_global_flag = generation_circles
			}
			remove_global_flag = generation_circles
			set_global_flag = generation_all

			limit = {
				has_global_flag = generation_all
			}
			remove_global_flag = generation_all
			set_global_flag = generation_clusters
		}
	}
}
